public class libraryapp {
    
    public String bookname {get;set;}
    public String author {get;set;}
    public String result {get;set;}
       
    public libraryapp()
    {
        bookname = '';
		author = '';
		result = '';
    }
    
    public void savebook()
    {
        result = 'Book is saved!'+'Book name is : '+bookname+'Book author is : '+author;
    }
    
}

## Apex code here --> 

<apex:page controller="libraryapp">
    <apex:form>
    	<h1> Libray Application </h1>

    	<apex:pageBlock title="Add New Book">
            <apex:pageBlockSection columns="1">
                <apex:inputText value="{!bookname}" label="Book Name"/>
                <apex:inputText value="{!author}" label="Author"/>
                <apex:commandButton value="Save Book" action="{!savebook}" rerender="outputpanel"/>
            </apex:pageBlockSection>
    
    		
    	</apex:pageBlock>
        <apex:outputPanel id="outputpanel">
        	<h3>
                {!result}
            </h3>
        </apex:outputPanel>
    
    </apex:form>
	

</apex:page>
